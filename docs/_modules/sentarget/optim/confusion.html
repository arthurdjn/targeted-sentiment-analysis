r"""
Defines a ```ConfusionMatrix```, used to compute scores (True Positive, False Negative etc.).

.. image:: images/confusion_matrix.png


Example:

.. code-block:: python

    # Create a confusion matrix and ignore the labels
    # associated to <unk> and <pad>.
    confusion = ConfusionMatrix(num_classes=10, unk_idx=0, pad_idx=1)

    # Update the confusion matrix with a list of predictions and labels
    confusion.update(gold_labels, predictions)

    # Get the global accuracy, precision, scores from attributes or methods
    confusion.accuracy()

"""

[docs]class ConfusionMatrix:
    r"""A ```ConfusionMatrix``` is a matrix of shape :math:`(C, C)`, used to index predictions :math:`p \in C`
    regarding their gold labels (or truth labels).

    """

    def __init__(self, num_classes=None, data=None, ignore_index=None, labels=None):
        assert num_classes is not None or data is not None, 'Failed to initialize a confusion matrix. Please provide ' \
                                                            'either the number of classes `num_classes` or a starting ' \
                                                            'data `data`.'
        self.num_classes = num_classes if num_classes is not None else len(data)
        ignore_index = [ignore_index] if isinstance(ignore_index, int) else ignore_index
        self.ignore_index = [] if ignore_index is None else ignore_index
        self.labels = labels

        # Number of ignored elements
        gold_labels = np.arange(self.num_classes)
        indices = np.delete(gold_labels, self.ignore_index)
        self.matrix = np.zeros((indices.size, indices.size)) if data is None else np.array(data)
        self.label2idx = {label: key for (key, label) in enumerate(indices)}

    @property
    def tp(self):
        return np.diag(self.matrix)

    @property
    def tn(self):
        return self.matrix.sum() - (self.fp + self.fn + self.tp)

    @property
    def fp(self):
        return self.matrix.sum(axis=0) - np.diag(self.matrix)

    @property
    def fn(self):
        return self.matrix.sum(axis=1) - np.diag(self.matrix)

    @property
    def tpr(self):
        top = self.tp
        bottom = self.tp + self.fn
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def tnr(self):
        top = self.tn
        bottom = self.tn + self.fp
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def ppv(self):
        top = self.tp
        bottom = self.tp + self.fp
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def npv(self):
        top = self.tn
        bottom = self.tn + self.fn
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def fpr(self):
        top = self.fp
        bottom = self.fp + self.fn
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def fnr(self):
        top = self.fn
        bottom = self.tp + self.fn
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def fdr(self):
        top = self.fp
        bottom = self.tp + self.fp
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    @property
    def acc(self):
        top = self.tp + self.tn
        bottom = self.tp + self.tn + self.fn + self.tn
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

    def true_pos(self):
        return np.mean(self.tp)

    def true_neg(self):
        return np.mean(self.tn)

    def false_pos(self):
        return np.mean(self.fp)

    def false_neg(self):
        return np.mean(self.fn)

    def accuracy(self):
        return np.mean(self.acc)

    def sensitivity(self):
        return np.mean(self.tpr)

    def precision(self):
        return np.mean(self.ppv)

    def specificity(self):
        return np.mean(self.tnr)


[docs]    def update(self, gold_labels, predictions):
        r"""Update the confusion matrix from a list of predictions, with their respective gold labels.

        Args:
            gold_labels (list): a list of predictions.
            predictions (list): respective gold labels (or truth labels)

        """
        # Make sure the inputs are 1D arrays
        gold_labels = np.array(gold_labels).reshape(-1)
        predictions = np.array(predictions).reshape(-1)
        # Complete the confusion matrix
        for i, p in enumerate(predictions):
            # Ignore unknown predictions or pad index
            if (gold_labels[i] not in self.ignore_index and
                    predictions[i] not in self.ignore_index):
                actual = self.label2idx[gold_labels[i]]
                pred = self.label2idx[predictions[i]]
                self.matrix[actual, pred] += 1



[docs]    def to_dataframe(self, labels=None, normalize=False):
        r"""Convert the ``ConfusionMatrix`` to a `DataFrame`.

        Args:
            labels (list): list containing the ordered names of the indices used as gold labels.
            normalize (bool): if ``True``, normalize the ``matrix``.

        Returns:
            pandas.DataFrame

        """
        labels = labels or self.labels
        matrix = self.normalize() if normalize else self.matrix
        return pd.DataFrame(matrix, index=labels, columns=labels)



[docs]    def to_dict(self):
        r"""Convert the ``ConfusionMatrix`` to a `dict`.

        * :attr:`global accuracy` (float): accuracy obtained on all classes.

        * :attr:`sensitivity` (float): sensitivity obtained on all classes.

        * :attr:`precision` (float): precision obtained on all classes.

        * :attr:`specificity` (float): specificity obtained on all classes.

        * :attr:`confusion` (list): confusion matrix obtained on all classes.

        Returns:
            dict

        """
        return {'global accuracy': float(self.accuracy()),
                'sensitivity': float(self.sensitivity()),
                'precision': float(self.precision()),
                'specificity': float(self.specificity()),
                'confusion': self.matrix.tolist()}



[docs]    def normalize(self):
        r"""Nomalize the confusion ``matrix``.

        .. math::

            \text{Norm}(Confusion) = \frac{Confusion}{sum(Confusion)}

        .. note::

            The operation is not inplace, and thus does not modify the attribute ```matrix```.


        Returns:
            numpy.ndarray: normalized confusion matrix.

        """
        top = self.matrix
        bottom = self.matrix.sum(axis=1)[:, np.newaxis]
        return np.divide(top, bottom, out=np.zeros_like(top), where=bottom != 0)

        r"""Zeros the ```matrix```. Can be used to empty memory without removing the object.

        Returns:
            None. Inplace operation.

        """
        self.matrix = np.zeros_like(self.matrix)
